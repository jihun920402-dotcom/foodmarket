);

-- 주문 정보를 저장하는 테이블
CREATE TABLE market_orders (
    order_id NUMBER PRIMARY KEY,
    userid VARCHAR2(50) REFERENCES market_members(userid),
    p_name VARCHAR2(100), -- 대표 상품명 (ex: 전복 외 2건)

-- 마일리지 충전 신청 테이블
CREATE TABLE market_charge_request (
    request_id NUMBER PRIMARY KEY,          -- 신청 고유번호
    userid VARCHAR2(50) NOT NULL,           -- 신청자 아이디
    amount NUMBER NOT NULL,                 -- 신청 금액
    status VARCHAR2(20) DEFAULT 'pending',  -- 상태 (pending: 대기, success: 승인)
    request_date DATE DEFAULT SYSDATE,      -- 신청 날짜
    CONSTRAINT fk_charge_user FOREIGN KEY (userid) REFERENCES market_members(userid)
);

-- 신청 번호 자동 생성을 위한 시퀀스
CREATE SEQUENCE charge_seq START WITH 1 INCREMENT BY 1;

COMMIT;

DROP SEQUENCE order_item_seq;

-- 2. 주문 메인 테이블 (누가, 언제, 어디로, 총 얼마를)
CREATE TABLE market_orders (
    order_id NUMBER PRIMARY KEY,            -- 주문 고유번호
    userid VARCHAR2(50) NOT NULL,           -- 주문자 ID
    total_price NUMBER NOT NULL,            -- 총 결제 금액
    order_date DATE DEFAULT SYSDATE,        -- 주문 일시
    receiver_name VARCHAR2(50),             -- 수령인 이름
    receiver_phone VARCHAR2(20),            -- 수령인 연락처
    address VARCHAR2(200),                  -- 배송지 주소
    status VARCHAR2(20) DEFAULT '결제완료',   -- 상태 (결제완료/배송중/배송완료/취소됨)
    CONSTRAINT fk_order_user FOREIGN KEY (userid) REFERENCES market_members(userid)
);

-- 3. 주문 상세 테이블 (무엇을, 몇 개, 개당 얼마에 샀는가)
-- 이 테이블이 있어야 나중에 상품별 판매 순위나 정확한 재고 관리가 가능합니다.
CREATE TABLE market_order_items (
    item_id NUMBER PRIMARY KEY,             -- 상세내역 고유번호
    order_id NUMBER NOT NULL,               -- 연결된 주문번호
    p_id NUMBER NOT NULL,                   -- 구매한 상품번호
    count NUMBER NOT NULL,                  -- 구매 수량
    order_price NUMBER NOT NULL,            -- 구매 당시 가격
    CONSTRAINT fk_items_order FOREIGN KEY (order_id) REFERENCES market_orders(order_id),
    CONSTRAINT fk_items_product FOREIGN KEY (p_id) REFERENCES market_products(p_id)
);

-- 4. 시퀀스 생성
CREATE SEQUENCE order_seq START WITH 1000 INCREMENT BY 1;      -- 주문번호는 1000번부터 시작
CREATE SEQUENCE order_item_seq START WITH 1 INCREMENT BY 1;

-- 5. 변경사항 저장
COMMIT;

CREATE TABLE market_reviews (
    r_id          NUMBER PRIMARY KEY,      -- 리뷰 고유 번호
    p_id          NUMBER NOT NULL,         -- 상품 번호 (FK)
    userid        VARCHAR2(50) NOT NULL,   -- 작성자 아이디 (FK)
    content       VARCHAR2(1000) NOT NULL, -- 리뷰 내용
    rating        NUMBER(1) DEFAULT 5,     -- 별점 (1~5)
    r_date        DATE DEFAULT SYSDATE,    -- 작성일
    CONSTRAINT fk_review_p_id FOREIGN KEY (p_id) REFERENCES market_products(p_id),
    CONSTRAINT fk_review_userid FOREIGN KEY (userid) REFERENCES market_members(userid)
);

CREATE SEQUENCE review_seq START WITH 1 INCREMENT BY 1;

commit;


package common;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.SQLException;

public class DBConnection {
	public static Connection getConnection() {
		Connection conn = null;

		// 연결 정보 (SQL Developer에서 성공했던 정보와 동일!)
		String driver = "oracle.jdbc.driver.OracleDriver";
		String url = "jdbc:oracle:thin:@localhost:1521/FREEPDB1"; // 서비스 이름 주의!
		String user = "system";
		String pass = "1234";

		try {
			// 1. 드라이버 로드
			Class.forName(driver);
			// 2. 연결 생성
			conn = DriverManager.getConnection(url, user, pass);
			System.out.println("✅ 해산물 DB 연결 성공!");
		} catch (ClassNotFoundException e) {
			System.out.println("❌ 드라이버 로드 실패 (ojdbc 파일 확인 필요)");
			e.printStackTrace();
		} catch (SQLException e) {
			System.out.println("❌ DB 연결 실패 (URL/ID/PW 확인 필요)");
			e.printStackTrace();
		}

		return conn;
	}
}
